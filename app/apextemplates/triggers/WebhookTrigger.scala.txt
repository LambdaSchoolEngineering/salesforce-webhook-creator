@(name: String, sobject: String, events: List[String], url: String)trigger @{name}WebhookTrigger on @{sobject} (@{events.mkString(",")}) {

    String url = '@url';

    String eventTime = Trigger.isBefore ? 'before' : 'after';
    String eventName = Trigger.isInsert ? 'insert' : Trigger.isUpdate ? 'update' : Trigger.isDelete ? 'delete' : Trigger.isUndelete ? 'undelete' : 'unknown';

    String content = Webhook.jsonContent(Trigger.new, Trigger.old, eventName + '.' + eventTime);

    Webhook.callout(url, content);

}
